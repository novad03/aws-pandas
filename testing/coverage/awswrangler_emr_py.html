486 def build_step(self, 

487 name: str, 

488 command: str, 

489 action_on_failure: str = "CONTINUE", 

490 script: bool = False) -> Dict[str, Collection[str]]: 

491 """ 

492 Build the Step dictionary 

493 :param name: Step name 

494 :param command: e.g. 'echo "Hello!"' | e.g. for script 's3://.../script.sh arg1 arg2' 

495 :param action_on_failure: 'TERMINATE_JOB_FLOW', 'TERMINATE_CLUSTER', 'CANCEL_AND_WAIT', 'CONTINUE' 

496 :param script: True for raw command or False for script runner (https://docs.aws.amazon.com/emr/latest/ReleaseGuide/emr-commandrunner.html) 

497 :return: Step Dict 

498 """ 

499 jar: str = "command-runner.jar" 

500 if script is True: 

501 if self._session.region_name: 

502 region: str = self._session.region_name 

503 else: 

504 region = "us-east-1" 

505 jar = f"s3://{region}.elasticmapreduce/libs/script-runner/script-runner.jar" 

506 step = { 

507 "Name": name, 

508 "ActionOnFailure": action_on_failure, 

509 "HadoopJarStep": { 

510 "Jar": jar, 

511 "Args": command.split(" ") 

512 } 

513 } 

514 return step
