307 explode_outer: bool = True, 

308 explode_pos: bool = True) -> List[Tuple[str, str, str]]: 

309 explode: str = "EXPLODE_OUTER" if explode_outer is True else "EXPLODE"

314 cols += Spark._flatten_struct_column(path=path, dtype=dtype) 

315 elif Spark._is_array(dtype=dtype):

317 elif Spark._is_map(dtype=dtype): 

318 cols.append((path, "map")) 

319 else:

322 expr: str 

323 for path, dtype in cols: 

324 path_under = path.replace('.', '_') 

325 if Spark._is_array(dtype): 

326 if explode_pos: 

327 expr = f"{explode}({path}) AS ({path_under}_pos, {path_under})" 

328 else: 

329 expr = f"{explode}({path}) AS {path_under}" 

330 elif Spark._is_map(dtype):

336 expr = f"{path} AS {path.replace('.', '_')}"

345 @staticmethod 

346 def flatten(dataframe: DataFrame, 

347 explode_outer: bool = True, 

348 explode_pos: bool = True, 

349 name: str = "root") -> Dict[str, DataFrame]: 

350 """ 

351 Convert a complex nested DataFrame in one (or many) flat DataFrames 

352 If a columns is a struct it is flatten directly. 

353 If a columns is an array or map, then child DataFrames are created in different granularities. 

354 :param dataframe: Spark DataFrame 

355 :param explode_outer: Should we preserve the null values on arrays? 

356 :param explode_pos: Create columns with the index of the ex-array 

357 :param name: The name of the root Dataframe 

358 :return: A dictionary with the names as Keys and the DataFrames as Values 

359 """ 

360 cols_exprs: List[Tuple[str, str, str]] = Spark._flatten_struct_dataframe(df=dataframe, 

361 explode_outer=explode_outer, 

362 explode_pos=explode_pos) 

363 exprs_arr: List[str] = [x[2] for x in cols_exprs if Spark._is_array_or_map(x[1])] 

364 exprs: List[str] = [x[2] for x in cols_exprs if not Spark._is_array_or_map(x[1])] 

365 dfs: Dict[str, DataFrame] = {name: dataframe.selectExpr(exprs)} 

366 exprs = [x[2] for x in cols_exprs if not Spark._is_array_or_map(x[1]) and not x[0].endswith("_pos")] 

367 for expr in exprs_arr: 

368 df_arr = dataframe.selectExpr(exprs + [expr]) 

369 name_new: str = Spark._build_name(name=name, expr=expr) 

370 dfs_new = Spark.flatten(dataframe=df_arr, 

371 explode_outer=explode_outer, 

372 explode_pos=explode_pos, 

373 name=name_new) 

374 dfs = {**dfs, **dfs_new} 

375 return dfs
